Aster RPC — Public API Reference

This module re-exports the public API surface for documentation. Import from aster directly in your code, not from aster.public.

Getting Started

Producer (server)::

from aster import AsterServer, service, rpc, wire_type
from dataclasses import dataclass

@wire_type("myapp/GreetRequest")
@dataclass
class GreetRequest:
    name: str = ""

@wire_type("myapp/GreetResponse")
@dataclass
class GreetResponse:
    message: str = ""

@service(name="Greeter", version=1)
class Greeter:
    @rpc
    async def greet(self, req: GreetRequest) -> GreetResponse:
        return GreetResponse(message=f"Hello, {req.name}!")

async with AsterServer(services=[Greeter()]) as srv:
    print(srv.ticket)  # share this with consumers
    await srv.serve()

Consumer (client)::

from aster import AsterClient

client = AsterClient(address="aster1...")
await client.connect()

# Use the shell for exploration:
#   aster shell aster1...

# Or generate a typed client:
#   aster contract gen-client aster1... --out ./clients --package myapp

Decorators

Use these to define RPC services:

  • service() — Declare a class as an RPC service
  • rpc() — Mark a method as a unary RPC endpoint
  • server_stream() — Mark a method as server-streaming
  • client_stream() — Mark a method as client-streaming
  • bidi_stream() — Mark a method as bidirectional streaming
  • wire_type() — Register a dataclass for cross-language serialization

Error Handling

All RPC failures raise RpcError with a StatusCode.

  85class AsterServer:
  86    """High-level, declarative producer.
  87
  88    Builds a single :class:`IrohNode` that serves blobs + docs + gossip
  89    (iroh built-in protocols) alongside aster RPC (``aster/1``) and any
  90    enabled admission ALPNs — all on **one endpoint, one node ID**.
  91
  92    When any admission gate is active (``allow_all_consumers=False`` or
  93    ``allow_all_producers=False``), the node is built with
  94    ``enable_hooks=True`` and a background task runs the Gate 0
  95    connection-level hook loop (``MeshEndpointHook.run_hook_loop``), which
  96    gates *all* protocols (blobs, docs, gossip, aster/1, admission) at the
  97    QUIC handshake layer.
  98    """

 252    async def start(self) -> None:
 253        """Create the unified node and compute ``ServiceSummary`` list. Idempotent."""
 254        if self._started:
 255            return
 256
 257        # Configure structured logging from config (idempotent)
 258        from aster.logging import configure_logging
 259        configure_logging(
 260            format=self._config.log_format,
 261            level=self._config.log_level,
 262            mask=self._config.log_mask,
 263        )
 264
 265        # Determine which aster ALPNs to register on the Router.
 266        # Consumer admission is ALWAYS registered — even in open-gate mode
 267        # the consumer uses it to discover services.
 268        aster_alpns: list[bytes] = [RPC_ALPN, ALPN_CONSUMER_ADMISSION]
 269        gate0_needed = False
 270        if not self._allow_all_consumers:
 271            if self._hook is None:
 272                self._hook = MeshEndpointHook()
 273            if self._nonce_store is None:
 274                self._nonce_store = InMemoryNonceStore()
 275            gate0_needed = True
 276        if not self._allow_all_producers:
 277            aster_alpns.append(ALPN_PRODUCER_ADMISSION)
 278            if self._hook is None:
 279                self._hook = MeshEndpointHook()
 280            gate0_needed = True
 281
 282        # Build EndpointConfig so hooks (Gate 0) are installed when needed.
 283        ep_cfg = _build_node_endpoint_config(
 284            self._endpoint_config_template, enable_hooks=gate0_needed
 285        )
 286
 287        self._node = await IrohNode.memory_with_alpns(aster_alpns, ep_cfg)
 288        addr_b64 = base64.b64encode(
 289            self._node.node_addr_info().to_bytes()
 290        ).decode()
 291
 292        # Auto-create ephemeral MeshState. Even when allow_all_producers=True
 293        # we need the topic_id so the root node's shell can observe gossip.
 294        if self._mesh_state is None and self._root_pubkey is not None:
 295            self._mesh_state = make_ephemeral_mesh_state(self._root_pubkey)
 296
 297        # Build ServiceSummary list with per-spec contract_id.
 298        summaries: list[ServiceSummary] = []
 299        for svc in self._services_in:
 300            svc_cls = svc if inspect.isclass(svc) else type(svc)
 301            info = getattr(svc_cls, "__aster_service_info__", None)
 302            if info is None:
 303                raise TypeError(
 304                    f"{svc_cls!r} is not @service-decorated "
 305                    f"(missing __aster_service_info__)"
 306                )
 307            cid = contract_id_from_service(svc_cls)
 308            summaries.append(
 309                ServiceSummary(
 310                    name=info.name,
 311                    version=info.version,
 312                    contract_id=cid,
 313                    channels={self._channel_name: addr_b64},
 314                )
 315            )
 316        self._service_summaries = summaries
 317
 318        # ── C11.4.3: Startup publication verification ────────────────────
 319        # If .aster/manifest.json exists, verify that each service's live
 320        # contract_id matches the committed manifest. Fatal on mismatch.
 321        manifest_path = os.path.join(os.getcwd(), ".aster", "manifest.json")
 322        if os.path.isfile(manifest_path):
 323            import json as _json
 324            from .contract.manifest import ContractManifest
 325
 326            try:
 327                with open(manifest_path, encoding="utf-8") as f:
 328                    manifest_data = _json.load(f)
 329            except Exception as exc:
 330                raise RuntimeError(
 331                    f"Failed to read manifest at {manifest_path}: {exc}"
 332                ) from exc
 333
 334            # Support both single manifest and list of manifests
 335            manifests_list = manifest_data if isinstance(manifest_data, list) else [manifest_data]
 336            manifest_by_key: dict[tuple[str, int], ContractManifest] = {}
 337            for md in manifests_list:
 338                m = ContractManifest(**md)
 339                manifest_by_key[(m.service, m.version)] = m
 340
 341            for summary in summaries:
 342                key = (summary.name, summary.version)
 343                manifest = manifest_by_key.get(key)
 344                if manifest is None:
 345                    continue  # Service not in manifest — skip
 346                if summary.contract_id != manifest.contract_id:
 347                    raise RuntimeError(
 348                        f"Contract identity mismatch for {summary.name!r} "
 349                        f"v{summary.version}:\n"
 350                        f"  Expected (manifest): {manifest.contract_id}\n"
 351                        f"  Actual (live):       {summary.contract_id}\n"
 352                        f"  Manifest: {manifest_path}\n"
 353                        f"  -> The service interface has changed without "
 354                        f"updating the manifest.\n"
 355                        f"     Rerun `aster contract gen` and commit the "
 356                        f"updated manifest."
 357                    )
 358            logger.debug("Manifest verification passed for %d service(s)", len(manifest_by_key))
 359
 360        # ── Contract publication to registry doc ─────────────────────────
 361        # Create a registry doc, publish contract collections (with manifest
 362        # including method schemas), and generate a read-only share ticket
 363        # so consumers can discover full contract metadata.
 364        await self._publish_contracts()
 365
 366        # ── Gate 3: capability interceptor & default-deny warning ────────
 367        #
 368        # Build a service_name -> ServiceInfo map for the CapabilityInterceptor
 369        # and check whether any service has authorization configured.
 370        from .interceptors.capability import CapabilityInterceptor
 371
 372        svc_map: dict[str, Any] = {}
 373        any_has_requires = False
 374        for svc in self._services_in:
 375            svc_cls = svc if inspect.isclass(svc) else type(svc)
 376            info = getattr(svc_cls, "__aster_service_info__", None)
 377            if info is not None:
 378                svc_map[info.name] = info
 379                if info.requires is not None:
 380                    any_has_requires = True
 381                for mi in info.methods.values():
 382                    if mi.requires is not None:
 383                        any_has_requires = True
 384
 385        # Auto-wire the CapabilityInterceptor when trust is configured
 386        # (Gate 0 enabled) or when any service declares requires.
 387        has_auth_interceptor = any(
 388            isinstance(i, CapabilityInterceptor) for i in self._interceptors
 389        )
 390        if (not self._allow_all_consumers or any_has_requires) and not has_auth_interceptor:
 391            # Prepend so capability is checked before application interceptors.
 392            self._interceptors.insert(0, CapabilityInterceptor(svc_map))
 393
 394        # S12.2: default-deny startup warning.
 395        # When Gate 0 is disabled (allow_all_consumers=True), any service
 396        # without explicit authorization is wide open.  Emit a warning so
 397        # the developer knows.
 398        if self._allow_all_consumers:
 399            for svc in self._services_in:
 400                svc_cls = svc if inspect.isclass(svc) else type(svc)
 401                info = getattr(svc_cls, "__aster_service_info__", None)
 402                if info is None:
 403                    continue
 404                if info.public:
 405                    continue
 406                svc_has_auth = info.requires is not None
 407                if not svc_has_auth:
 408                    svc_has_auth = any(
 409                        mi.requires is not None for mi in info.methods.values()
 410                    )
 411                if not svc_has_auth:
 412                    # Also check if user explicitly added an auth interceptor.
 413                    from .interceptors.auth import AuthInterceptor
 414                    has_explicit_auth = any(
 415                        isinstance(i, AuthInterceptor) for i in self._interceptors
 416                    )
 417                    if not has_explicit_auth:
 418                        warnings.warn(
 419                            f"Service '{info.name}' has no authorization configured "
 420                            f"and Gate 0 is disabled (allow_all_consumers=True). "
 421                            f"All consumers can call this service without "
 422                            f"authentication. Add @service(requires=...) or "
 423                            f"configure an auth interceptor for production use.",
 424                            UserWarning,
 425                            stacklevel=2,
 426                        )
 427
 428        # Server borrows a NetClient view of the node. AsterServer owns the
 429        # node lifecycle, so Server must NOT close the endpoint on its own.
 430        self._server = Server(
 431            net_client(self._node),
 432            services=self._services_in,
 433            codec=self._codec,
 434            interceptors=self._interceptors,
 435            owns_endpoint=False,
 436            peer_store=self._peer_store,
 437        )
 438
 439        self._print_banner()
 440        self._started = True

 442    def _print_banner(self) -> None:

 503            mode_parts.append(f"{G}registry{R}")
 504        w(f"    {D}mode:{R}      {' '.join(mode_parts)}\n")
 505

 508        log_level = os.environ.get("ASTER_LOG_LEVEL", "info")

 515        except Exception:

 526    async def _publish_contracts(self) -> None:
 527        """Create a registry doc and publish each service's contract collection.
 528
 529        After publication, ``self._registry_namespace`` is set to the 64-char
 530        hex namespace_id so consumer admission can return it.
 531        """
 532        assert self._node is not None
 533
 534        try:
 535            dc = docs_client(self._node)
 536            bc = blobs_client(self._node)
 537
 538            # Create the registry doc (producer owns the write capability)
 539            registry_doc = await dc.create()
 540            author_id = await dc.create_author()
 541
 542            for svc in self._services_in:
 543                svc_cls = svc if inspect.isclass(svc) else type(svc)
 544                info = getattr(svc_cls, "__aster_service_info__", None)
 545                if info is None:
 546                    continue
 547
 548                # Build the type graph and contract
 549                from .contract.identity import (
 550                    ServiceContract,
 551                    build_type_graph,
 552                    canonical_xlang_bytes,
 553                    resolve_with_cycles,
 554                    compute_type_hash,
 555                )
 556                from .contract.publication import build_collection, upload_collection
 557                from .registry.keys import contract_key, version_key
 558                from .registry.models import ArtifactRef
 559
 560                # Collect root types
 561                root_types: list[type] = []
 562                for mi in info.methods.values():
 563                    if mi.request_type is not None:
 564                        root_types.append(mi.request_type)
 565                    if mi.response_type is not None:
 566                        root_types.append(mi.response_type)
 567
 568                type_graph = build_type_graph(root_types)
 569                type_defs = resolve_with_cycles(type_graph)
 570
 571                # Compute type hashes
 572                type_hashes: dict[str, bytes] = {}
 573                for fqn, td in type_defs.items():
 574                    td_bytes = canonical_xlang_bytes(td)
 575                    type_hashes[fqn] = compute_type_hash(td_bytes)
 576
 577                # Build ServiceContract and canonical bytes
 578                contract = ServiceContract.from_service_info(info, type_hashes)
 579                contract_bytes = canonical_xlang_bytes(contract)
 580
 581                import blake3 as _blake3
 582                contract_id = _blake3.blake3(contract_bytes).hexdigest()
 583
 584                # Build collection with full method schemas
 585                entries = build_collection(contract, type_defs, service_info=info)
 586
 587                # Upload to blob store as a native HashSeq collection.
 588                # GC protection is handled automatically by the HashSeq tag.
 589                collection_hash = await upload_collection(bc, entries)
 590
 591                # Create a collection ticket so consumers can download all
 592                # collection blobs (index + entries) in one transfer
 593                blob_ticket = bc.create_collection_ticket(collection_hash)
 594
 595                # Write ArtifactRef to registry doc
 596                import time as _time
 597                ref = ArtifactRef(
 598                    contract_id=contract_id,
 599                    collection_hash=collection_hash,
 600                    ticket=blob_ticket,
 601                    published_by=author_id,
 602                    published_at_epoch_ms=int(_time.time() * 1000),
 603                    collection_format="index",
 604                )
 605                await registry_doc.set_bytes(
 606                    author_id,
 607                    contract_key(contract_id),
 608                    ref.to_json().encode(),
 609                )
 610
 611                # Also write the manifest JSON directly to the registry doc
 612                # at a well-known key. This avoids the blob download round-trip
 613                # for consumers that only need method schemas (like the shell).
 614                manifest_data = None
 615                for ename, edata in entries:
 616                    if ename == "manifest.json":
 617                        manifest_data = edata
 618                        break
 619                if manifest_data:
 620                    from .registry.keys import version_key as _vk
 621                    manifest_key = f"manifests/{contract_id}".encode()
 622                    await registry_doc.set_bytes(
 623                        author_id, manifest_key, manifest_data
 624                    )
 625
 626                # Version pointer
 627                await registry_doc.set_bytes(
 628                    author_id,
 629                    version_key(info.name, info.version),
 630                    contract_id.encode(),
 631                )
 632
 633                logger.debug(
 634                    "Published contract %s for %s v%d (collection=%s)",
 635                    contract_id[:12],
 636                    info.name,
 637                    info.version,
 638                    collection_hash[:12],
 639                )
 640
 641            # share() enables the sync engine on the server side so consumers
 642            # can replicate this doc.  We only need the namespace_id on the wire.
 643            await registry_doc.share_with_addr("read")
 644            self._registry_namespace = registry_doc.doc_id()
 645            logger.debug(
 646                "Registry doc ready — namespace: %s", self._registry_namespace[:16]
 647            )
 648
 649        except Exception as exc:
 650            # Publication failure is non-fatal — the server still works,
 651            # consumers just won't get rich contract metadata
 652            logger.warning("Contract publication failed (non-fatal): %s", exc)

 654    def serve(self) -> asyncio.Task:
 655        """Spawn the accept loop (+ Gate 0 hook loop); return an aggregate task.
 656
 657        ``await server.serve()`` blocks until cancellation. The second call
 658        returns the same task (idempotent).
 659        """
 660        if self._serve_task is not None:
 661            return self._serve_task
 662        if not self._started:
 663            raise RuntimeError("AsterServer.serve() called before start()")
 664        assert self._server is not None
 665        assert self._node is not None
 666
 667        subtasks: list[asyncio.Task] = []
 668
 669        # Gate 0 hook loop: drain the after-handshake channel, apply the
 670        # MeshEndpointHook allowlist for every connection. before_connect is
 671        # auto-accepted inside NodeHookReceiver (the peer's endpoint ID
 672        # isn't authenticated at that stage).
 673        if self._hook is not None and self._node.has_hooks():
 674            self._hook_loop_task = asyncio.create_task(
 675                self._run_gate0(), name="aster-gate0"
 676            )
 677            subtasks.append(self._hook_loop_task)
 678
 679        subtasks.append(
 680            asyncio.create_task(self._accept_loop(), name="aster-accept")
 681        )
 682
 683        # Delegated admission loop: accept connections on aster.admission
 684        # ALPN for @aster-issued enrollment tokens.
 685        if self._delegation_policies:
 686            subtasks.append(
 687                asyncio.create_task(
 688                    self._delegated_admission_loop(), name="aster-delegated-admission"
 689                )
 690            )
 691
 692        # Auto-register endpoints with @aster for published services.
 693        # Requires producer service tokens (from `aster publish`).
 694        if self._producer_tokens:
 695            subtasks.append(
 696                asyncio.create_task(
 697                    self._aster_registration_loop(), name="aster-registration"
 698                )
 699            )
 700
 701        self._subtasks = subtasks
 702
 703        async def _wait_all() -> None:
 704            try:
 705                await asyncio.gather(*subtasks, return_exceptions=True)
 706            except asyncio.CancelledError:
 707                # Graceful shutdown on Ctrl+C / task cancellation
 708                logger.info("Server shutting down...")
 709                for t in subtasks:
 710                    t.cancel()
 711                await asyncio.gather(*subtasks, return_exceptions=True)
 712
 713        self._serve_task = asyncio.create_task(_wait_all(), name="aster-server-serve")
 714        return self._serve_task
 715
 716    async def _run_gate0(self) -> None:
 717        """Take the hook receiver from the node and run the hook loop."""
 718        assert self._node is not None
 719        assert self._hook is not None
 720        receiver = await self._node.take_hook_receiver()
 721        if receiver is None:
 722            logger.warning("AsterServer: hooks enabled but no receiver available")
 723            return
 724        await self._hook.run_hook_loop(receiver)

 726    async def _accept_loop(self) -> None:
 727        """Pull from ``node.accept_aster()`` and dispatch per ALPN."""
 728        assert self._node is not None
 729        assert self._server is not None
 730        services_snapshot = list(self._service_summaries)
 731        try:
 732            while True:
 733                try:
 734                    alpn, conn = await self._node.accept_aster()
 735                except asyncio.CancelledError:
 736                    raise
 737                except Exception as exc:  # noqa: BLE001
 738                    logger.warning("AsterServer: accept_aster failed: %s", exc)
 739                    continue
 740
 741                if alpn == RPC_ALPN:
 742                    asyncio.create_task(
 743                        self._server.handle_connection(conn),
 744                        name="aster-rpc-conn",
 745                    )
 746                elif alpn == ALPN_CONSUMER_ADMISSION:
 747                    # Always handle consumer admission — even when
 748                    # allow_all_consumers=True the consumer needs the
 749                    # services list from the admission response.
 750                    asyncio.create_task(
 751                        handle_consumer_admission_connection(
 752                            conn,
 753                            root_pubkey=self._root_pubkey or b"\x00" * 32,
 754                            hook=self._hook,
 755                            nonce_store=self._nonce_store,
 756                            services_getter=lambda: services_snapshot,
 757                            registry_namespace_getter=lambda: self._registry_namespace,
 758                            allow_unenrolled=self._allow_all_consumers,
 759                            peer_store=self._peer_store,
 760                            gossip_topic_getter=lambda: (
 761                                self._mesh_state.topic_id
 762                                if self._mesh_state else None
 763                            ),
 764                        ),
 765                        name="aster-consumer-admission-conn",
 766                    )
 767                elif alpn == ALPN_PRODUCER_ADMISSION and not self._allow_all_producers:
 768                    assert self._root_pubkey is not None
 769                    assert self._mesh_state is not None
 770                    asyncio.create_task(
 771                        handle_producer_admission_connection(
 772                            conn,
 773                            own_root_pubkey=self._root_pubkey,
 774                            own_state=self._mesh_state,
 775                            config=self._clock_drift_config,
 776                            persist_state=self._persist_mesh_state,
 777                        ),
 778                        name="aster-producer-admission-conn",
 779                    )
 780                else:
 781                    try:
 782                        conn.close(0, b"unexpected alpn")
 783                    except Exception:  # noqa: BLE001
 784                        pass
 785        except asyncio.CancelledError:
 786            pass

 788    async def drain(self, grace_period: float = 10.0) -> None:
 789        """Graceful shutdown: stop accepting new connections, drain existing ones.
 790
 791        Compatible with Kubernetes ``preStop`` hooks and SIGTERM handling.
 792        After drain completes, call ``close()`` to shut down the node.
 793
 794        Args:
 795            grace_period: Seconds to wait for in-flight requests to complete.
 796        """
 797        logger.info("Draining server (grace_period=%.1fs)...", grace_period)
 798        if self._server is not None:
 799            await self._server.drain(grace_period)
 800        logger.info("Drain complete")

 802    async def close(self) -> None:
 803        """Cancel serve loops and close the node. Safe to call multiple times."""
 804        if self._closed:
 805            return
 806        self._closed = True
 807
 808        for t in self._subtasks:
 809            t.cancel()
 810        if self._serve_task is not None:
 811            self._serve_task.cancel()
 812            try:
 813                await self._serve_task
 814            except (asyncio.CancelledError, Exception):
 815                pass
 816
 817        # Close the node — this triggers router.shutdown() which closes all
 818        # protocol handlers (including aster queue handlers) and the endpoint.
 819        if self._node is not None:
 820            try:
 821                await self._node.close()
 822            except Exception:
 823                pass

 914        """Background loop: register endpoints with @aster for published services.
 915
 916        Connects to @aster, registers each service's endpoint, then
 917        re-registers periodically before TTL expiry.
 918        """
 919        import json as _json
 920
 921        # Wait a moment for the server to be fully ready
 922        await asyncio.sleep(2)
 923
 924        ttl = 300  # 5 minutes
 925        interval = ttl * 0.75  # re-register at 75% of TTL
 926

 931                return

 933                logger.warning("@aster registration failed: %s", exc)
 934
 935            # Wait before re-registering
 936            try:
 937                await asyncio.sleep(interval)
 938            except asyncio.CancelledError:
 939                return

 941    async def _register_endpoints_with_aster(self, ttl: int) -> None:
 942        """One-shot registration of all published service endpoints."""
 943        if not self._node or not self._producer_tokens:
 944            return
 945
 946        # Build our endpoint info
 947        addr_info = self._node.node_addr_info()
 948        node_id = addr_info.endpoint_id
 949        relay = addr_info.relay_url or ""
 950        direct_addrs = addr_info.direct_addresses or []
 951
 952        # Resolve @aster address from identity file or profile config.
 953        # The token itself contains the root_pubkey which can be used
 954        # to discover @aster via DNS TXT record in production.
 955        aster_addr = self._resolve_aster_address()
 956        if not aster_addr:
 957            logger.debug("No @aster address configured — skipping registration")
 958            return
 959
 960        aster_client = AsterClient(address=aster_addr)
 961
 962        try:
 963            await aster_client.connect()
 964        except Exception as exc:
 965            logger.debug("Could not connect to @aster: %s", exc)
 966            return
 967
 968        try:
 969            # For each published service with a token, call register_endpoint
 970            for svc_name, token in self._producer_tokens.items():
 971                try:
 972                    # Use the dynamic invoke path — we don't have generated
 973                    # types for @aster's PublicationService
 974                    import json as _json
 975                    request = {
 976                        "producer_token": _json.dumps(token),
 977                        "node_id": node_id,
 978                        "relay": relay,
 979                        "direct_addrs": direct_addrs,
 980                        "ttl": ttl,
 981                    }
 982
 983                    # Invoke register_endpoint on PublicationService
 984                    conn = await aster_client._rpc_conn_for(
 985                        next(
 986                            (s.channels.get("rpc", "") for s in aster_client._services
 987                             if s.name == "PublicationService"),
 988                            ""
 989                        )
 990                    )
 991                    from .transport.iroh import IrohTransport
 992                    transport = IrohTransport(conn, codec=self._codec)
 993                    resp = await transport.unary(
 994                        "PublicationService", "register_endpoint", request
 995                    )
 996                    logger.info(
 997                        "Registered endpoint with @aster: %s (%s)",
 998                        svc_name, node_id[:12],
 999                    )
1000                except Exception as exc:
1001                    logger.warning(
1002                        "Failed to register %s with @aster: %s",
1003                        svc_name, exc,
1004                    )
1005        finally:
1006            await aster_client.close()

1009        """Resolve the @aster service address for endpoint registration.
1010
1011        Checks (in order):
1012        1. ASTER_SERVICE_ADDRESS env var
1013        2. aster_service.address in the identity file's peer entry
1014        3. DNS TXT record on aster.site (future)

1032    def _install_signal_handlers(self, grace_period: float = 10.0) -> None:
1033        """Install SIGTERM/SIGINT handlers for graceful shutdown.
1034
1035        Call after ``serve()`` to enable k8s-compatible shutdown:
1036
1037        - SIGTERM: drain → close (graceful)
1038        - SIGINT (Ctrl+C): drain → close (graceful)
1039        - Second SIGINT: immediate exit
1040
1041        Usage::
1042
1043            async with AsterServer(services=[...]) as srv:
1044                srv.install_signal_handlers()
1045                await srv.serve()
1046        """
1047        import signal
1048
1049        loop = asyncio.get_event_loop()
1050        shutdown_count = 0
1051
1052        def _handle_signal(sig: int, frame: Any) -> None:
1053            nonlocal shutdown_count
1054            shutdown_count += 1
1055            if shutdown_count > 1:
1056                logger.warning("Forced exit (second signal)")
1057                sys.exit(1)
1058            logger.info("Received %s — draining...", signal.Signals(sig).name)
1059            loop.create_task(self._graceful_shutdown(grace_period))
1060
1061        signal.signal(signal.SIGTERM, _handle_signal)
1062        signal.signal(signal.SIGINT, _handle_signal)

1065        """Internal: drain then close."""
1066        try:
1067            await self.drain(grace_period)
1068        finally:
1069            await self.close()

444    def load_identity(self, peer_name: str | None = None,
445                      role: str | None = None) -> tuple[bytes | None, dict | None]:
446        """Load a peer entry from the ``.aster-identity`` file.
447
448        Args:
449            peer_name: Select peer by ``name``.  If None, auto-selects by
450                ``role`` (first match).
451            role: Fallback selector when ``peer_name`` is None.
452                ``"producer"`` for AsterServer, ``"consumer"`` for AsterClient.
453
454        Returns:
455            ``(secret_key_bytes, peer_dict)`` or ``(None, None)`` when no
456            identity file is found.
457        """
458        import base64 as _b64
459
460        path = self._resolve_identity_path()
461        if path is None:
462            return None, None
463
464        if sys.version_info >= (3, 11):
465            import tomllib as _tl
466        else:
467            import tomli as _tl  # type: ignore[no-redef]
468
469        with open(path, "rb") as f:
470            data = _tl.load(f)
471
472        node = data.get("node", {})
473        secret_key_b64 = node.get("secret_key")
474        secret_key = _b64.b64decode(secret_key_b64) if secret_key_b64 else None
475
476        # Find the peer entry
477        peers = data.get("peers", [])

479        if peer_name is not None:
480            for p in peers:
481                if p.get("name") == peer_name:
482                    peer = p
483                    break
484        elif role is not None:
485            for p in peers:
486                if p.get("role") == role:
487                    peer = p
488                    break
489        elif peers:
490            peer = peers[0]  # single peer → auto-select
491
492        return secret_key, peer

Trust model (Aster-trust-spec.md §1.1): The root private key is offline — it never touches a running node.     config = AsterConfig.from_env()

    config = AsterConfig.from_file("aster.toml")

  3. Inline (testing, scripts)::

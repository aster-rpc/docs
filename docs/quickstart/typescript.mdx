# TypeScript Quickstart

Build a peer-to-peer RPC service in under 5 minutes. By the end you'll have a server and client talking over the Aster protocol.

## Prerequisites

- Node.js 20+ or Bun 1.0+
- TypeScript 5.0+

## 1. Install

```bash
# bun (recommended — faster)
bun init
bun add @aster-rpc/aster

# or npm
npm init -y
npm install @aster-rpc/aster
```

## 2. Define your types

Create `types.ts`:

```typescript
import { WireType } from '@aster-rpc/aster';

@WireType("hello/HelloRequest")
export class HelloRequest {
  name = "";
  constructor(init?: Partial<HelloRequest>) {
    if (init) Object.assign(this, init);
  }
}

@WireType("hello/HelloResponse")
export class HelloResponse {
  message = "";
  constructor(init?: Partial<HelloResponse>) {
    if (init) Object.assign(this, init);
  }
}
```

The `@WireType` tag is the stable wire identity for cross-language serialization. A Python client using `@wire_type("hello/HelloRequest")` will produce compatible bytes.

## 3. Define your service

Create `service.ts`:

```typescript
import { Service, Rpc, ServerStream } from '@aster-rpc/aster';
import { HelloRequest, HelloResponse } from './types.js';

@Service({ name: "Hello", version: 1 })
export class HelloService {
  @Rpc({ timeout: 30, idempotent: true })
  async sayHello(req: HelloRequest): Promise<HelloResponse> {
    return new HelloResponse({
      message: `Hello, ${req.name}!`,
    });
  }

  @ServerStream()
  async *countdown(req: HelloRequest): AsyncGenerator<HelloResponse> {
    for (let i = 3; i > 0; i--) {
      yield new HelloResponse({
        message: `${req.name}: ${i}...`,
      });
    }
    yield new HelloResponse({
      message: `${req.name}: Go!`,
    });
  }
}
```

## 4. Run the server

Create `server.ts`:

```typescript
import { ServiceRegistry, LocalTransport } from '@aster-rpc/aster';
import { HelloService } from './service.js';

const registry = new ServiceRegistry();
registry.register(new HelloService());

console.log("Hello service registered");

// For local testing, use LocalTransport (in-process, no network)
export const transport = new LocalTransport(registry);
```

## 5. Call from a client

Create `client.ts`:

```typescript
import { createClient } from '@aster-rpc/aster';
import { HelloRequest } from './types.js';
import { HelloService } from './service.js';
import { transport } from './server.js';

const client = createClient(HelloService, transport);

// Unary call
const result = await client.sayHello(new HelloRequest({ name: "World" }));
console.log(result.message);
// → "Hello, World!"

// Server streaming
for await (const item of client.countdown(new HelloRequest({ name: "TypeScript" }))) {
  console.log(item.message);
}
// → "TypeScript: 3..."
// → "TypeScript: 2..."
// → "TypeScript: 1..."
// → "TypeScript: Go!"

await client.close();
```

Run it:

```bash
bun run client.ts
```

## 6. Add interceptors

```typescript
import { ServiceRegistry, LocalTransport, MetricsInterceptor, DeadlineInterceptor } from '@aster-rpc/aster';
import { HelloService } from './service.js';

const metrics = new MetricsInterceptor();
const deadline = new DeadlineInterceptor();

// Interceptors are applied to the server dispatch chain
// (full interceptor wiring shown in the Server Reference)

console.log(metrics.snapshot());
// → { started: 0, succeeded: 0, failed: 0, inFlight: 0 }
```
